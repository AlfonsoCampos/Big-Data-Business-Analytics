Ejemplo: estructuras complejas

Vamos a crear una tabla que permita gestionar información del siguiente tipo:

create table empleados (
nombre STRING,
salario FLOAT,
subordinados ARRAY<STRING>,
deducciones MAP<STRING, FLOAT>,
direccion STRUCT<calle:STRING, ciudad:STRING, comunidad:STRING, codpostal:INT>);

1. Crear tabla que permita gestionar tipos de datos básicos

create table empleados1 (
nombre STRING,
salario FLOAT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ’,’
LINES TERMINATED BY ‘\n’
STORED AS TEXTFILE;

sudo nano /tmp/ejemplos/hive/empleados1

juan perez, 30.000
maria jimenez, 25.000
jesus sanchez, 20.000

hive -e "LOAD DATA LOCAL INPATH '/tmp/ejemplos/hive/empleados1' INTO TABLE empleados1";

hive -e "select nombre from empleados1;"

2. Crear una tabla que permita gestionar los tipos de datos ARRAY

create table empleados2 (
nombre STRING,
salario FLOAT,
subordinados ARRAY<STRING>
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ’,’
COLLECTION ITEMS TERMINATED BY ’#’
LINES TERMINATED BY ‘\n’
STORED AS TEXTFILE;

sudo nano /tmp/ejemplos/hive/empleados2

juan perez, 30.000, maria jimenez#jesus sanchez
maria jimenez, 25.000, jesus sanchez
jesus sanchez, 20.000

hive -e "LOAD DATA LOCAL INPATH '/tmp/ejemplos/hive/empleados2' INTO TABLE empleados2";
hive -e "select salario from empleados2;"
hive -e "select subordinados from empleados2";

3. Crear una tabla que permita gestionar los MAPS

create table empleados3 (
nombre STRING,
salario FLOAT,
subordinados ARRAY<STRING>,
deducciones MAP<STRING, FLOAT>
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ’,’
COLLECTION ITEMS TERMINATED BY ’#’
MAP KEYS TERMINATED BY ’:’
LINES TERMINATED BY ‘\n’
STORED AS TEXTFILE;

sudo nano /tmp/ejemplos/hive/empleados3

juan perez, 30.000,maria jimenez#jesus sanchez,IRPF:.25#SEG SOC:.05
maria jimenez, 25.000,jesus sanchez,IRPF:.2#SEG SOC:.05
jesus sanchez, 20.000,,IRPF:.22#SEG SOC:.05

hive -e "LOAD DATA LOCAL INPATH '/tmp/ejemplos/hive/empleados3' INTO TABLE empleados3";
hive -e "select nombre from empleados3 where salario > 20.000;"
hive -e "select deducciones from empleados3;"

4. Crear una tabla que permita crear todos los tipos de datos complejos

create table empleados4 (
nombre STRING,
salario FLOAT,
subordinados ARRAY<STRING>,
deducciones MAP<STRING, FLOAT>,
direccion STRUCT<calle:STRING, ciudad:STRING, comunidad:STRING, codpostal:INT>
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ’,’
COLLECTION ITEMS TERMINATED BY ’#’
MAP KEYS TERMINATED BY ’:’
LINES TERMINATED BY ‘\n’
STORED AS TEXTFILE;

sudo nano /tmp/ejemplos/hive/empleados4

juan perez, 30.000, maria jimenez#jesus sanchez,IRPF:.25#SEG SOC:.05,Gran Via#Madrid#Madrid#28001
maria jimenez, 25.000, jesus sanchez,IRPF:.2#SEG SOC:.05,Calle Mayor 12#Alpedrete#Madrid#28512
jesus sanchez, 20.000,,IRPF:.2#SEG SOC:.05,Calle Arenal 3#Madrid#Madrid#28001


hive -e "LOAD DATA LOCAL INPATH '/tmp/ejemplos/hive/empleados4' INTO TABLE empleados4";
hive -e "select nombre from empleados4 where salario > 20.000 and salario < 30.000;"
hive -e "select nombre, subordinados from empleados4 group by nombre;"
hive -e "select deducciones from empleados4;"
hive -e "select direccion.calle from empleados4;"
